DayNum=$( TZ=UTC date | cut -d' ' -f  4 | sort )
Time=$( TZ=UTC date | cut -d' ' -f  5 | sort )
Year=$( TZ=UTC date | cut -d' ' -f 7 | sort )
Name=cigelni2.txt
Underscore=_
Filename="$Year-$Month-$DayNum$Underscore$Time$Underscore$Name"

cp hw2.2script.sh $Filename
